#!/bin/bash

下载安装包的服务器地址

BASE_SERVER=192.168.2.2
yum install -y wget
wget $BASE_SERVER/software/jdk-8u161-linux-x64.tar.gz
tar -zxvf jdk-8u161-linux-x64.tar.gz -C /usr/share

配置环境变量

cat >> /etc/profile << EOF
export JAVA_HOME=/usr/share/jdk1.8.0_161
export PATH=\$PATH:\$JAVA_HOME/bin
EOF

```
