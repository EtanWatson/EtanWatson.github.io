     #如果BeautifulSoup对象里面没有nonExistentTag标签

 print(bsObj.nonExistentTag.someTag)
 #会抛出 AttributeError: 'NoneType' object has no attribute 'someTag'

 try:
     badContent = bsObj.nonExistingTag.anotherTag

     print("Tag was not found")

     if badContent == None:

         print(badContent)
